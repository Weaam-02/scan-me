public class App {

    
    //Write a static method named "findDistance" that calculates and returns
    //the distance between two points (x1,y1) and (x2,y2)
    //All parameters and return value are of type double
    //Parameters list should be match the following  (x1, y1, x2, y2)

    //Hint: to calculate the distance use the following equation:
    //distance = squareroot [(x2-x1)^2+(y2-y1)^2]
    //Note: you may use methods of Math class

    //Example: If point one is (7.5, 29.3) and point two is (48.1, 55.3)
    //Then, the return value should be: 62.7152294


    /**findDistance Method code should be below this comment**/
    public static double findDistance(double x1, double x2, double y1, double y2) {
        return Math.sqrt( Math.pow(x2 - x1, 2) + Math.pow(y2 - y1, 2) );
    }

    /****************End of findDistnace Method****************/


    //write a static method named "sumOfThreeMultiples" that 
    //takes one integer number as parameter 
    //and returns the sum of the first 3 multiples of that number

    //For example: if the number equals 7, then 7 + 14 + 21= 42
    //the return value should be: 42

    /**sumOfThreeMultiples Method code should be below this comment**/
    public static int sumOfThreeMultiples(int x) {
        return x + x*2 + x*3;
    }

    /***************End of sumOfThreeMultiples Method*****************/
    


    //write a static method named "findMaxDigit" that 
    //takes one integer number as a parameter which consists of 4 digits
    //and returns the maximum digit among the 4 

    //For example: if the given number is 2583
    //Then, the return value should be 8

    //Hint: to pull a number a part to its didgits, use n/10 and n%10
    //Note: use if-else to find the maximum digit

    // NEVER XcQ
    /**findMaxDigit Method code should be below this comment**/
    public static int findMaxDigit(int x) {

        // 2583 / 1000 = 2
        int firstDigit = x / 1000;

        // 2583 / 100 = 25, 25 % 10 = 5
        int secondDigit = (x / 100) % 10;

        // 2583 / 10 = 258, 258 % 10 = 8
        int thirdDigit = (x / 10) % 10;

        // 2583 % 10 = 3
        int fourthDigit = x % 10;

        // assume the first digit is the max digit
        int max = firstDigit;

        if (secondDigit > max) {
            max = secondDigit;
        }

        if (thirdDigit > max) {
            max = thirdDigit;
        }

        if (fourthDigit > max) {
            max = fourthDigit;
        }

        return max;
    }


    /***************End of findMaxDigit Method*****************/	

    
    //write a static method named "guessingNumber" that 
    //takes one integer number as a parameter between the range of 1 to 100
    //and returns true if the number matches a random number and displays "Congratulations!"
    //else method returns false if the number is less than the random number and displays "Too low!"
    //otherwise the method reutrns false if the number is greater than the random number and displays "Too high!"
    
    //For example: if the number equals 25 and the random number is 90
    //the return value should be false
    //and the output is: Too low!
    
    /** Hint: 
        * Use Random class methods of the `rng` variable defined below
        * For example, to get a random integer number between 0 and 10, and store it in a variable `r`,
        * you can use the following statement: 
        * int r = rng.nextInt(11);
        * 
        * Note 01: the Random object/variable has beed defined for you, so you only need to use it inside the method's body
        *          its called "rng"
        * Note 02: use if else to compare numbers
        */ 
    
    public static Random rng = new Random ();
    /**guessingNumber Method code should be below this comment**/
    public static boolean guessingNumber(int x) {

        int randomNumber = rng.nextInt(101);

        if (x > randomNumber) {
            System.out.println("Too high!");
            return false;
        }
        else if (x < randomNumber) {
            System.out.println("Too low!");
            return false;
        }
        else if (x == randomNumber) {
            System.out.println("Congratulations!");
            return true;
        }

        return false;
    }

    /***************End of guessingNumber Method*****************/


    //write a static method named "convertMonth" that takes a Gregorian month name
    //and returns the month number or -1 if the month name is wrong
    
    //For example: if the month is January, the returned value is 1	

    //Hint: you may use switch-cases

    /**convertMonth Method code should be below this comment**/
    public static int convertMonth(String monthName) {

        switch (monthName) {
            case "January":
                return 1;
            case "February":
                return 2;
            case "March":
                return 3;
            case "April":
                return 4;
            case "May":
                return 5;
            case "June":
                return 6;
            case "July":
                return 7;
            case "August":
                return 8;
            case "September":
                return 9;
            case "October":
                return 10;
            case "November":
                return 11;
            case "December":
                return 12;
            default:
                return -1;
        }
    }

    /***************End of convertMonth Method*****************/


    /*************** main Method *****************/

    public static void main (String []args){

        Scanner scanner = new Scanner(System.in);

        //prompt users to enter the coordinates of the first point x1 and y1 separated by space
        System.out.println("Enter first points x, y: ");
