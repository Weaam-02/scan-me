    public static boolean isMagicalWord(String str) {
        String reversed = "";

        for (int i = str.length() - 1; i >= 0; i--) {
            reversed += str.charAt(i);
        }

        return str.compareToIgnoreCase(reversed) == 0;
    }

    public static String wordsGame(String word1, String word2) {
        String sentence = "Go to the shop";

        if (word1.compareTo("Go") == 0) {
            return "No insertion at the beginning of the sentence!";

        int index = sentence.indexOf(word1);
        if (index == -1) {
            return word1 + " could not be found in the sentence";
        }
        return sentence.substring(0, index) + word1 + " " + word2 + sentence.subSequence(index + word1.length(), sentence.length());
